E2E Step 1: Start infrastructure

# Go to project root
cd /Users/thirapongpinkaew/Desktop/ME/SOC-Lab-AI-Intelligence-Layer

# Ensure shared Docker network exists
docker network inspect soc_shared >/dev/null 2>&1 || docker network create soc_shared

# Start Kafka stack
docker compose -f kafana/docker-compose.yml up -d

# Start RabbitMQ only (from OpenCTI stack)
docker compose -f opencti/docker-compose.yml up -d opencti-rabbitmq

# Check container status
docker ps --format 'table {{.Names}}\t{{.Status}}\t{{.Ports}}'

# Ensure topic exists
docker exec kafana-kafka-broker-1 kafka-topics \
  --bootstrap-server localhost:9092 \
  --create --if-not-exists \
  --topic pa5220.llm_output \
  --partitions 1 \
  --replication-factor 1
