# Go to project root
cd /Users/thirapongpinkaew/Desktop/ME/SOC-Lab-AI-Intelligence-Layer

# Create E2E queues and bindings
RABBIT_PASS=$(grep '^RABBITMQ_DEFAULT_PASS=' opencti/.env | cut -d= -f2-)

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -u thirapongp -p "$RABBIT_PASS" \
  declare queue name=e2e.alert.high durable=false

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -u thirapongp -p "$RABBIT_PASS" \
  declare queue name=e2e.alert.medium durable=false

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -u thirapongp -p "$RABBIT_PASS" \
  declare queue name=e2e.alert.low durable=false

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -u thirapongp -p "$RABBIT_PASS" \
  declare binding source=soc.events destination_type=queue destination=e2e.alert.high routing_key=alert.high

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -u thirapongp -p "$RABBIT_PASS" \
  declare binding source=soc.events destination_type=queue destination=e2e.alert.medium routing_key=alert.medium

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -u thirapongp -p "$RABBIT_PASS" \
  declare binding source=soc.events destination_type=queue destination=e2e.alert.low routing_key=alert.low

# Verify Kafka consumer group

docker exec grafana-kafka-broker-1 kafka-consumer-groups \
  --bootstrap-server localhost:9092 \
  --describe --group dispatcher-group

# Verify RabbitMQ queues

docker exec opencti-opencti-rabbitmq-1 rabbitmqctl list_queues name messages_ready messages_unacknowledged | egrep 'e2e.alert.high|e2e.alert.medium|e2e.alert.low'

# Peek messages from queues (requeue true)

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -f pretty_json -u thirapongp -p "$RABBIT_PASS" \
  get queue=e2e.alert.high count=1 ackmode=ack_requeue_true

docker exec opencti-opencti-rabbitmq-1 rabbitmqadmin -f pretty_json -u thirapongp -p "$RABBIT_PASS" \
